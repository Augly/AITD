### 接口设计

```python
class ExecutionBackend(ABC):
    def __init__(self, settings: dict[str, Any]) -> None

    @abstractmethod
    def sync_book(self, book) -> (book, warnings)
    @abstractmethod
    def apply_position_action(self, book, position, action, decision_id) -> (book, actions, warnings)
    @abstractmethod
    def open_position(self, book, candidate, entry, decision_id) -> (book, actions, warnings)
    @abstractmethod
    def apply_circuit_breaker(self, book, positions, decision_id) -> (book, actions, warnings)
    @abstractmethod
    def flatten_all_positions(self, book, decision_id) -> (book, actions, warnings)
```
